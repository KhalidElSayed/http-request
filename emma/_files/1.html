265                 * <p>
266                 * Encodes up to three bytes of the array <var>source</var> and writes
267                 * the resulting four Base64 bytes to <var>destination</var>. The source
268                 * and destination arrays can be manipulated anywhere along their length
269                 * by specifying <var>srcOffset</var> and <var>destOffset</var>. This
270                 * method does not check to make sure your arrays are large enough to

274                 * <var>numSigBytes</var>.
275                 * </p>
276                 * <p>

278                 * parameters.

282                 *            the array to convert

284                 *            the index where conversion begins
285                 * @param numSigBytes
286                 *            the number of significant bytes in your array
287                 * @param destination
288                 *            the array to hold the conversion
289                 * @param destOffset
290                 *            the index where output will be put
291                 * @return the <var>destination</var> array
292                 * @since 1.3
293                 */

295                                int numSigBytes, byte[] destination, int destOffset) {
296

298 
299                        int inBuff = (numSigBytes > 0 ? ((source[srcOffset] << 24) >>> 8)
300                                        : 0)
301                                        | (numSigBytes > 1 ? ((source[srcOffset + 1] << 24) >>> 16)
302                                                        : 0)
303                                        | (numSigBytes > 2 ? ((source[srcOffset + 2] << 24) >>> 24)
304                                                        : 0);
305 
306                        switch (numSigBytes) {
307                        case 3:
308                                destination[destOffset] = ALPHABET[(inBuff >>> 18)];
309                                destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
310                                destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
311                                destination[destOffset + 3] = ALPHABET[(inBuff) & 0x3f];
312                                return destination;
313 
314                        case 2:
315                                destination[destOffset] = ALPHABET[(inBuff >>> 18)];
316                                destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
317                                destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
318                                destination[destOffset + 3] = EQUALS_SIGN;
319                                return destination;
320 
321                        case 1:
322                                destination[destOffset] = ALPHABET[(inBuff >>> 18)];
323                                destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
324                                destination[destOffset + 2] = EQUALS_SIGN;
325                                destination[destOffset + 3] = EQUALS_SIGN;

327 
328                        default:
329                                return destination;

409                public static byte[] encodeBytesToBytes(byte[] source, int off, int len) {
410 
411                        if (source == null)
412                                throw new NullPointerException("Cannot serialize a null array.");
413 
414                        if (off < 0)
415                                throw new IllegalArgumentException(
416                                                "Cannot have negative offset: " + off);
417 
418                        if (len < 0)
419                                throw new IllegalArgumentException(
420                                                "Cannot have length offset: " + len);
421 
422                        if (off + len > source.length)
423                                throw new IllegalArgumentException(

425                                                                "Cannot have offset of %d and length of %d with array of length %d",
426                                                                off, len, source.length));
427 
428                        // Bytes needed for actual encoding
429                        int encLen = (len / 3) * 4 + (len % 3 > 0 ? 4 : 0);
430 
431                        byte[] outBuff = new byte[encLen];
432 
433                        int d = 0;
434                        int e = 0;
435                        int len2 = len - 2;
436                        for (; d < len2; d += 3, e += 4)
437                                encode3to4(source, d + off, 3, outBuff, e);

441                                e += 4;
442                        }
443 
444                        if (e <= outBuff.length - 1) {
445                                byte[] finalOut = new byte[e];
446                                System.arraycopy(outBuff, 0, finalOut, 0, e);
447                                return finalOut;
448                        } else
449                                return outBuff;
450                }
